Linux LPIC Cheat-Sheet

tl;dr

Add User

# Create a new user with a home directory and set the default shell to bash
sudo useradd -m [-G audio,video] -s /bin/bash newuser

# Set or update the password for the newly created user
sudo passwd newuser
# or expire current password to let user set their own
sudo passwd -e newuser

# Optional: Add the user to the sudo group for administrative privileges
sudo usermod -aG sudo newuser

Partition and Mount File

# Open the disk utility to partition a drive (e.g., the second disk /dev/sdb)
sudo fdisk /dev/sdb
# Open a help [m]enu:
<m>
# Print the [p]artition table:
<p>
# Add a [n]ew partition:
<n>
# Select a partition to [d]elete:
<d>
# [w]rite table to disk and exit:
<w>
# [q]uit without saving changes:
<q>

# Format the new partition (e.g., /dev/sdb1) with the ext4 file system
sudo mkfs.ext4 /dev/sdb3

# Create a directory that will serve as the mount point
sudo mkdir -p /mnt/data_drive

# Mount the formatted partition to the newly created directory
sudo mount /dev/sdb3 /mnt/data_drive

Search files

# Search for a file by its exact name starting from the root directory
find / -name "syslog"

# Find all files (type f) in the home directory that are larger than 100 Megabytes
find /home -type f -size +100M

# Search for a specific text string inside all files within a directory (recursive, show line number, ignore-case)
grep -rni "error" /var/log/ 2> /dev/null

Zip, Unzip

# Compress Decompress
gzip file.txt # replace file.txt with file.txt.gz use -k to keep the original
gzip -d file.txt.gz # decompress

# create tarball
tar cvf files.tar files/* another_file.txt...

# Create a standard Linux tarball (.tar.gz) for better compression
tar -czvf backup_archive.tar.gz /path/to/folder
# Extract a tarball (.tar.gz) into a specific destination directory
tar -xzvf backup_archive.tar.gz -C /path/to/destination

Setup automount in /etc/fstab

# Retrieve the unique UUID of the partition (crucial for stable mounting)
sudo blkid /dev/sdb1

# Open the fstab configuration file in a text editor
sudo vim /etc/fstab

# Append this line to the end of the file (replace with your actual UUID):
# UUID=1234abcd-12ab-34cd-56ef-1234567890ab /mnt/data_drive ext4 defaults 0 2

# Test the fstab file to ensure there are no syntax errors before rebooting
sudo mount -a

STDIN, STDOUT, STDERR

COMMAND > FILE  # Makes a new or overwrites stdout to a file
COMMAND >> FILE # Makes a new or appends stdout to a file
# Same for STDERR
COMMAND 2> FILE
COMMAND 2>> FILE

# write STDOUT & STDERR in same file
COMMAND > FILE 2>&1
COMMAND >> FILE 2>&1 # STDERR behaves the same

# File as STDIN
COMMAND < FILE
COMMAND <<EOF
some text with multible
lines that don\'t have
to be from a file
EOF

One liner

# make parent directories as needed
mkdir -p a/couple/of/dirs

# jump back to last directory
cd -

# remove directories and their contents recursively
rm -r a/couple/of/dirs

# Count files recursive
find . -type f | wc -l

# Switch to root without changing startuser environment
su root
sudo bash

# Switch to root with changing startuser environment
su -      # With root password
sudo -i   # With user password

# list all dirs that mach the pattern "foo" from `.`
find -type d -name '*foo*'
# Add suffix `.txt` to every file in dir recursive
find . -type f -exec mv {} {}.txt \;

# case instentive
[...] | grep -i PATTERN
# list all files in `/etc` that contains the pattern username
grep username /etc/* 2> /dev/null

# create multiple files at once (file1a, file1b, file2a,...) 
touch file{1,2,3,4}{a,b}
touch file-{a..z}

ls file?a # any char between e and a
ls file?[a,b] # any char between e and (a or b)
ls file[^1-3]a # all file that don't have the number 1-3 
ls file[[:digit:]][[:alpha:]]

# execute second command if true
test -e states.txt && echo "File exists"

# grep only uncommend setting (no leading # or blank lines)
grep -v '^#' /etc/default/keyboard | grep -v '^$'
egrep -v ^'(#|$)' /etc/default/keyboard # with regex

# display resulting command expression
echo COMMAND
echo rm file{1..4} # -> rm file1 file2 file3 file4

# Show diff between two commands side-by-side with colors
diff --side-by-side --color=always <(ls -1 /proc/ -I "[[:digit:]]*" | sort) <(ls -1 /proc/1 | sort)

Filemanagement

# Make directory with needed parrentdir
mkdir -p a/bunch/of/dirs

# dd - convert and copy a file
# input file `/dev/zero`, output file `zero.txt`
# read  and  write  up to 1Mb at a time
# copy only 100 input blocks
dd if=/dev/zero of=zero.txt bs=1M count=100 # result a 100Mb large file with zeros
# Backup entire disk to disk
sudo dd if=/dev/sda of=/dev/sdb bs=4096 conv=noerror,sync status=progress
# Creating a compressed disk image
sudo dd if=/dev/vda status=progress | gzip -c > /tmp/vdadisk.img.gz
# restore
sudo gzip -dc /tmp/vdadisk.img.gz | dd of=/dev/vda # gunzip would work the same

# tar - an archiving utility

Commands

# Search Replace `help :substitute`
:[range]s[ubstitute]/{pattern}/{string}/[flags] [count] 
# Examples
# Confirm each from enteir file
:s/old/new/gc
# Line range from 3 to 6
:3,6s/old/new/g
# from current line to end of file
:.,$s/old/new/g
# Relative lines
:+2,-2/old/new/g
# Enteir file
:%s/old/new/g
# First in current line
:s/old/new/
# Ignor case
:s/old/new/gi

# Comment and uncomment from 5 to 20
:5,20s/^/#/
:5,20s/^#//
# Replace Multiple words
:%s/apple\|orange\|mango/fruit/g

# Insert the file below the cursor
:read log.txt
# Insert the stdout from the given command
:read! echo $USER

# copy last register into system clipboard
:let @+=@"

# Switch to Hexeditor
:%!xxd
# Go back/Apply changes from the hex part
:%!xxd -r
# Set filtetype (syntax hilighting) to xxd
:set filetype=xxd

# Set foldmethod
:set foldmethod=[manual|indent|syntax|diff|...]

Usermanagement

# id - print real and effective user and group IDs
id [OPTION]... [USER]...

# add user foobaa with home-dir (m) and Group (G) audio,video and shell (s) `/bin/bash`
useradd -mG audio,video -s /bin/bash foobaa
# get passwd entry from foobaa
getent passwd foobaa
# set password for foobaa
passwd foobaa
# expire current password for foobaa
passwd -e foobaa

# Print info about user account
passwd -S foobaa
cage -l foobaa

# block an account
passwd -l foobaa
# unblock an account
passwd -u foobaa

# Create Group foogroup with gid 10000
groupadd -g 10000 foogroup
# Add foobaa to group foogroup
usermod -aG foogroup foobaa

# permissions (1=x, 2=w, 4=r)
# add execute permission to group and user
chmod u+x,g+x file.sh
chmod +110 file.sh
# all permission user no write to else
chmod u=xwr,go=xr file.sh
chmod 755 file.sh

# Remove a user
userdel username

# Remove a user along with the home directory and mail spool
sudo userdel --remove username

# Prints the current umask
umask [-S]
# Add read right to all in umask
umask a+r

Prozesse, Sockets & System Demonds

# run command as job
COMMAND &
# list all jobs
jobs
# kill job (see `man signal`)
kill [-SIG_NUM] %JOB_ID

# list all pid with given name
pidof bash
pgrep bash
# kill process
kill [-SIG_NUM] PROZESS_ID

# process snapshot
# current terminal processes
ps
# list with "nice" value
ps -o pid,ni,cmd
# user processes
ps x
# all proxesses
ps aux
# as a hierarchy
ps fax

# Creating a Unix Socket
# SOCK_STREAM
nc --unixsock /tmp/demo-tcp.sock --listen
# SOCK_DGRAM
nc --unixsock /tmp/demo-udp.sock --udp --listen

# connect to to listen socken
# written text will be sendet bidirectional
nc --unixsock /tmp/demo-tcp.sock
nc --unixsock /tmp/demo-udp.sock -u

# Sending message to journalctl
logger "msg"
nc -U /run/sytemd/journal/dev-log -u

# See journalctl
journalctl -f

# start process with certain nice value
# -20 (most favorable to the process (only with root privilages)) to 19 (least favorable to the process)
# default 10
nice -NICE_VALUE COMAND

# systemctl - Control the systemd system and service manager
# show all running services
systemctl status
# Start/Stop/Restart/Reload/Show the status of a service:
systemctl start|stop|restart|reload|status UNIT
# Show the contents & absolute path of a unit file or edit it:
systemctl cat|edit UNIT
# Enable/Disable a unit to be started on bootup:
systemctl enable|disable UNIT
# Reload systemd, scan for new or changed units:
systemctl daemon-reload
# Check if a unit is active/enabled/failed:
systemctl is-active|is-enabled|is-failed UNIT
# List all service/socket/automount units filtering by running/failed state:
systemctl list-units --type service|socket|automount --state failed|running

# set default target to boot into
systemctl set-default TARGET # e.g. graphical.target, multi-user.target, rescue.target,...
# switch to a target
systemctl isolate TARGET
# switch to default target
systemctl default

# Display the dependency tree for a specific target
systemctl list-dependencies graphical.target
